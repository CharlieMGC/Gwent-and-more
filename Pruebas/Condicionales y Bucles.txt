public class Test
{
    public static void Main()
    {
        // Prueba si `n` funciona correctamente como parámetro no predefinido
        int result = sumOfSquares(5); // Debería devolver 55 (1^2 + 2^2 + 3^2 + 4^2 + 5^2)
        Console.WriteLine(result); // Verifica el resultado en la consola
    }

    public static int sumOfSquares(int n)
    {
        int sum = 0;
        for (int i = 1; i <= n; i++)
        {
            sum = sum + i * i;
        }
        return sum;
    }
}
